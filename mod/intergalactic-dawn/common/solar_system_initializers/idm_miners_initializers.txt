			}

		generate_home_system_resources = yes

			limit = {
				is_inside_nebula = no

				radius = 25
				effect = { nebula_cloaking_effect = yes }
			}
		}
	}

	# Star

		orbit_distance = 0

				days = -1
			}

	}
